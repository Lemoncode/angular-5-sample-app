```diff typescript
+const nameValid = (control: FormControl): { [key: string]: any } => {
+  const firstLetter = control.value.toString()[0];
+  return (!!firstLetter && (firstLetter !== firstLetter.toUpperCase())) ?
+    { 'nameValid' : 'invalid name' } : null;
+};
```
